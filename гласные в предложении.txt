        public static int glasniye(int kolvo)
        {
            string predlozheniye = Console.ReadLine();
            char[] letters = new char[] { 'à', 'ó', 'å', 'î', 'ý', 'ÿ', 'è', 'û' };
            foreach (char letter in predlozheniye)
            {
               foreach(char let in letters)
                {
                    if (let == letter)
                    {
                        kolvo++;
                    }                                    
                }                   
            }

            return kolvo;
        }

        public static int factorial(int x)
        {
            if (x == 1)
            {
                return x;
            }
            else
            {
                return x * factorial(x-1);
            }
        }

        public static double task(double a,double b,double c=0.0f)
